4️⃣ Work Item Validation Script

Create: scripts/validate-workitem.py

import os, requests, sys

ORG="yourorg"
PROJECT="yourproject"
PAT=os.environ["AZURE_DEVOPS_EXT_PAT"]
WORKITEM_ID=os.environ["WORKITEM_ID"]

url=f"https://dev.azure.com/{ORG}/{PROJECT}/_apis/wit/workitems/{WORKITEM_ID}?api-version=7.0"

r=requests.get(url,auth=("",PAT))
data=r.json()["fields"]

if data.get("Custom.UATSignoff")!="Yes":
    sys.exit("UAT Signoff missing")

if not data.get("Custom.TestClasses"):
    sys.exit("Test classes mandatory")

print("Work item validation passed")
